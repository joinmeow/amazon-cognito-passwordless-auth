function normalizeEndpoint(endpoint: string): string {
  // Ensure the endpoint has a protocol and no trailing slash
  const withProtocol = endpoint.startsWith("http")
    ? endpoint
    : `https://${endpoint}`;
  return withProtocol.replace(/\/+$/, ""); // trim trailing slashes
}
